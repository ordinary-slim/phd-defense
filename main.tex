      \label{eq:extended_temp}
      T_e(\mathbf{x}, t) =
      \begin{cases}
        T(\mathbf{x}, t) & \mathbf{x} \in \overline{\Omega}(t)\\
        T_{dep} & \mathbf{x} \in \overline{\Omega}(t_{final}) \setminus \overline{\Omega}(t)
      \end{cases}
      \qquad
      f_{l,\; e}(\mathbf{x}, t) = f_l(T_e(\mathbf{x}, t))
    \end{equation}
    \begin{align}
      \label{eq:original_pde}
      \rho c_p \partial_t T_e + \rho L \partial_t f_{l,\; e} - k \Delta T
      &= r(\mathbf{x}, t) &&\forall \mathbf{x} \in \Omega(t)\\
      - k \partial_n T &= h_{conv} (T - T_{\text{env}}) + \varepsilon \sigma (T^4 - T_{\text{env}}^4) &&\forall \mathbf{x} \in \partial \Omega(t)\\
      \label{eq:initial_condition}
      T(\mathbf{x}, 0) &= T_0 \qquad &&\forall \mathbf{x} \in \Omega(0)
    \end{align}
