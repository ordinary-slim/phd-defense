% Set frame footer
\setbeamertemplate{frame footer}{EXPLICIT FRAME FOOTER TEXT}

% Custom footline that uses frame footer
\defbeamertemplate*{footline}{test}{%
  \leavevmode%
  \hbox{%
    \begin{beamercolorbox}[wd=.1\textwidth,left,sep=1.5ex]{footline}%
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=.8\textwidth,center,sep=1.5ex]{footline}%
      \usebeamertemplate*{frame footer}%
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=.1\textwidth,right,sep=1.5ex]{footline}%
      \insertframenumber%
    \end{beamercolorbox}%
  }%
}
