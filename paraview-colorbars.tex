\newcommand{\vcolorbar}[6]{
  \begin{tikzpicture}
    \pgfmathsetmacro{\myheight}{10*#2}
    \pgfmathsetmacro{\widthcolorbar}{#2}
    \pgfmathsetmacro{\cmin}{#3}
    \pgfmathsetmacro{\cmax}{#4}
    \def\extraticks{#6}
    \ifx\extraticks\empty
      \def\ticklist{0,1}
    \else
      \def\ticklist{0,\extraticks,1}
    \fi
    \begin{axis}[
      hide axis,
      scale only axis,
      height=\myheight,
      width=\widthcolorbar,
      colormap name={#1},
      colorbar,
      colorbar style={
        ytick={\ticklist},
        yticklabel pos=right,
        yticklabel style={font=\footnotesize},
        yticklabel={
          \pgfmathparse{\cmin+(\cmax-\cmin)*\tick}\pgfmathprintnumber[precision=1, fixed]{\pgfmathresult}
        },
        ylabel=#5,
        ylabel style={
          at={(0.0,0.5)},
          anchor=south,
          rotate=0,
          font=\footnotesize,
          overlay,
        },
        yticklabel style={
          font=\footnotesize,
          overlay,
        },
        width=\widthcolorbar,
      },
      point meta min=0,
      point meta max=1,
    ]
      % dummy plot just to draw colorbar:
      \addplot [draw=none] coordinates {(0,0) (0,1)};
    \end{axis}
  \end{tikzpicture}
}
